CREATE TABLE technician_profiles (
  id SERIAL PRIMARY KEY,
  technician_id INTEGER UNIQUE REFERENCES users(id),
  bio TEXT,
  location TEXT,
  skills TEXT[], -- Array of strings
  availability TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
